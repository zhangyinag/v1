<template>
  <div class="login">
    <el-form  label-width="100px">
      <el-form-item label="用户名">
        <el-input
          v-model="param.username"
          clearable
          placeholder="UM账号">
        </el-input>
      </el-form-item>
      <el-form-item label="密码">
        <el-input
          type="password"
          v-model="param.password"
          auto-complete="off"
          clearable
          placeholder="密码">
        </el-input>
      </el-form-item>
      <el-form-item>
        <el-button type="primary" @click="submitForm()" :disabled="submitting">登录</el-button>
      </el-form-item>
    </el-form>
  </div>
</template>
 
<script>
  import NProgress from 'nprogress' // Progress 进度条
  import 'nprogress/nprogress.css'// Progress 进度条样式
 
  const adminUser = {
    username: 'admin'
  }
 
  const tmrUser = {
    username: 'tmr'
  }
 
  export default {
    name: 'login',
    data () {
      return {
        submitting: false,
        param: {
          username: '',
          password: ''
        }
      }
    },
    methods: {
      submitForm () {
        NProgress.start()
        this.submitting = true
        setTimeout(() => {
          NProgress.done()
          this.submitting = false
          if (this.param.username === adminUser.username) {
            this.$router.push('/admin')
          } else if (this.param.username === tmrUser.username) {
            this.$router.push('/tmr')
          } else {
            this.$message.error('用户名或密码错误')
          }
        }, 3000)
      }
    }
  }
</script>
